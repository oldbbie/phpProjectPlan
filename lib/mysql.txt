id 날짜 계획명 실행여부

CREATE TABLE category (
    id INT NOT NULL AUTO_INCREMENT,
    name VARCHAR(30) NOT NULL,
    hide TINYINT NULL,
    PRIMARY KEY(id)
);

CREATE TABLE plan_name (
    id INT NOT NULL AUTO_INCREMENT,
    name VARCHAR(30) NOT NULL,
    content text NULL,
    next_con text NULL,
    hide TINYINT NULL,
    category_id int,
    PRIMARY KEY(id),
    FOREIGN KEY (category_id) REFERENCES category(id)
);


CREATE TABLE doit (
    id INT NOT NULL AUTO_INCREMENT,
    day DATE NOT NULL,
    yn TINYINT NULL,
    plan_name_id int,
    PRIMARY KEY(id),
    FOREIGN KEY (plan_name_id) REFERENCES plan_name(id)
);
